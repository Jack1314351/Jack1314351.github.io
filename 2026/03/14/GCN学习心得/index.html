GCN学习心得

1.核心公式


: 是大家当前的简历（特征）， 是一个翻译权重。每个人先把自己的原始简历交给公司（模型），公司用统一的眼光（）提取出最关键的信息。这步是特征转换，让简历变得更精简、更有针对性。

: 是带自环的邻接矩阵。每个人不仅看好朋友的简历，也得看看自己的。大家拉个群，把群里所有人的“简历精华”都收拢到一起。这步叫消息传递。

:对称归一化（Symmetric Normalization）。如果一个人有 1000 个朋友，他的声音会特别响；如果只有 1 个朋友，声音就很弱。为了公平，公式会根据你的“度”（朋友数量 ）做一个缩放：朋友越多的人，每句话的话语权就降低一点，防止“社交达人”掩盖了其他人的声音。

:激活函数（如 ReLU 或 tanh）。最后公司领导审核一遍，把那些没意义的杂讯过滤掉，只留下对分类真正有用的信息。

3.复现代码

import torch
import networkx as nx
import matplotlib.pyplot as plt
import time
from torch.nn import Linear
from torch_geometric.nn import GCNConv
from torch_geometric.datasets import KarateClub


# 1. 定义可视化函数
def visualize(h, color, epoch=None, loss=None, accuracy=None):
    plt.figure(figsize=(7, 7))
    plt.xticks([])
    plt.yticks([])

    if torch.is_tensor(h):
        h = h.detach().cpu().numpy()
        plt.scatter(h[:, 0], h[:, 1], s=140, c=color, cmap="Set2")
        if epoch is not None and loss is not None and accuracy is not None:
            plt.xlabel((f'Epoch: {epoch}, Loss: {loss.item():.4f} \n'
                        f'Train Acc: {accuracy["train"] * 100:.2f}%, '
                        f'Val Acc: {accuracy["val"] * 100:.2f}%'),
                       fontsize=12)
    else:
        nx.draw_networkx(h, pos=nx.spring_layout(h, seed=42), with_labels=False,
                         node_color=color, cmap="Set2")

    plt.show(block=False)  # 本地运行建议加 block=False，防止阻塞程序
    plt.pause(0.2)
    plt.close()


# 2. 加载数据
dataset = KarateClub()
data = dataset[0]


# 3. 定义 GCN 模型
class GCN(torch.nn.Module):
    def __init__(self):
        super().__init__()
        torch.manual_seed(1234)
        self.conv1 = GCNConv(dataset.num_features, 4)
        self.conv2 = GCNConv(4, 4)
        self.conv3 = GCNConv(4, 2)  # 最终降到 2 维，方便在平面画图
        self.classifier = Linear(2, dataset.num_classes)

    def forward(self, x, edge_index):
        h = self.conv1(x, edge_index)
        h = h.tanh()
        h = self.conv2(h, edge_index)
        h = h.tanh()
        h = self.conv3(h, edge_index)
        h = h.tanh()  # 节点嵌入向量

        out = self.classifier(h)
        return out, h


# 4. 初始化模型、损失函数和优化器
model = GCN()
criterion = torch.nn.CrossEntropyLoss()
optimizer = torch.optim.Adam(model.parameters(), lr=0.01)


# 5. 定义训练函数
def train(data):
    optimizer.zero_grad()
    out, h = model(data.x, data.edge_index)
    # 半监督学习：只根据 train_mask 的节点计算 Loss
    loss = criterion(out[data.train_mask], data.y[data.train_mask])
    loss.backward()
    optimizer.step()

    accuracy = {}
    # 计算训练集准确率
    predicted_classes = torch.argmax(out[data.train_mask], axis=1)
    target_classes = data.y[data.train_mask]
    accuracy['train'] = torch.mean((predicted_classes == target_classes).float()).item()

    # 计算全图准确率（验证集）
    predicted_classes = torch.argmax(out, axis=1)
    target_classes = data.y
    accuracy['val'] = torch.mean((predicted_classes == target_classes).float()).item()

    return loss, h, accuracy


# 6. 开始循环训练
print("模型结构如下：")
print(model)

for epoch in range(301):  # 跑 300 轮
    loss, h, accuracy = train(data)
    # 每 10 轮可视化一次
    if epoch % 10 == 0:
        print(f"Epoch {epoch:03d} | Loss: {loss:.4f} | Val Acc: {accuracy['val']:.2%}")
        visualize(h, color=data.y, epoch=epoch, loss=loss, accuracy=accuracy)

print("训练结束！")
